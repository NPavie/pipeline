subsequence($to,min((count($to),2))), concat( if (count($from) and string-length($from[1])) then '../' else '', if (count($to) > 1 and string-length($to[1])) then concat($to[1],'/') else '' ))"/> </xsl:otherwise> </xsl:choose> </xsl:function> </xsl:stylesheet>
