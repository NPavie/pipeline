207 	public VoiceManager(Collection<TTSEngine> engines, Collection<VoiceInfo> extraVoices) {
208 		//build the list of voices ordered by priority
209 		final float priorityVariantPenalty = 0.1f;
210 		Set<VoiceInfo> priorities = new LinkedHashSet<VoiceInfo>(BuiltinVoices);
211 		priorities.addAll(extraVoices);
212 		mVoicePriorities = new ArrayList<VoiceInfo>(priorities);
213 		for (VoiceInfo vinfo : priorities) {
214 			if (vinfo.language != null){
215 				String shortLang = vinfo.language.getLanguage();
216 				if (!vinfo.language.equals(new Locale(shortLang))) {
217 					try {
218 						mVoicePriorities.add(new VoiceInfo(vinfo.voice, shortLang, vinfo.gender,
219 						        vinfo.priority - priorityVariantPenalty));
220 					} catch (UnknownLanguage e1) {
221 						//should not happen
222 					}
223 				}
224 			}
225 		}
226 		Comparator<VoiceInfo> reverseComp = new Comparator<VoiceInfo>() {
227 			@Override
228 			public int compare(VoiceInfo v1, VoiceInfo v2) {
229 				return Float.valueOf(v2.priority).compareTo(Float.valueOf(v1.priority));
230 			}
231 		};
232 		Collections.sort(mVoicePriorities, reverseComp);
233 
234 		//Create a map of the best services for each available voice, given that two different
235 		//services can serve the same voice. 
236 		TTSTimeout timeout = new TTSTimeout();
237 		for (TTSEngine tts : engines) { //no possible concurrent write
238 			try {
239 				timeout.enableForCurrentThread(3);
240 				Collection<Voice> voices = tts.getAvailableVoices();
241 				if (voices != null)
242 					for (Voice v : voices) {
243 						TTSEngine competitor = mBestEngines.get(v);
244 						if (competitor == null
245 						        || competitor.getOverallPriority() < tts.getOverallPriority()) {
246 							mBestEngines.put(v, tts);
247 						}
248 					}
249 			} catch (SynthesisException e) {
250 				ServerLogger.error("error while retrieving the voices of "
251 				        + TTSServiceUtil.displayName(tts.getProvider()));
252 				ServerLogger.debug(TTSServiceUtil.displayName(tts.getProvider())
253 				        + " getAvailableVoices error: " + getStack(e));
254 			} catch (InterruptedException e) {
255 				ServerLogger.error("timeout while retrieving the voices of "
256 				        + TTSServiceUtil.displayName(tts.getProvider()));
257 			} finally {
258 				timeout.disable();
259 			}
260 		}
261 		timeout.close();
262 		
263 		//get the best voice that can handle any language, if any
264 		for (VoiceInfo voiceInfo : mVoicePriorities) {
265 			if (voiceInfo.isMultiLang() && mBestEngines.containsKey(voiceInfo.voice)){
266 				mBestMultiLangVoice = voiceInfo.voice;
267 				break;
268 			}
269 		}
270 		
271 		//find all the languages. That will help us construct a multi-lang voice
272 		HashSet<Locale> allLangs = new HashSet<Locale>();
273 		for (VoiceInfo voiceInfo : mVoicePriorities)
274 			allLangs.add(voiceInfo.language);
275 		allLangs.remove(VoiceInfo.NO_DEFINITE_LANG);
276 		
277 
278 		//Create maps for the best voices depending on the information available among
279 		//the language, the engine and the gender
280 		for (VoiceInfo voiceInfo : mVoicePriorities) {
281 			if (voiceInfo.isMultiLang()){
282 				for (Locale lang : allLangs)
283 					registerVoice(voiceInfo, lang);
284 			}
285 			else
286 				registerVoice(voiceInfo, voiceInfo.language);
287 		}
288 
289 		/*
290 		 * Create a map of the best fallback voices for each language. engine is
291 		 * more important than priority and gender. Gender is more important
292 		 * than priority.
293 		 */
294 		List<VoiceInfo> sortedVoices = new ArrayList<VoiceInfo>();
295 		for (int i = 0; i < mVoicePriorities.size(); ++i) {
296 			VoiceInfo vi = mVoicePriorities.get(i);
297 			if (mBestEngines.containsKey(vi.voice))
298 				sortedVoices.add(vi);
299 		}
300 
301 		setSecondVoices(sortedVoices, true, true);
302 		setSecondVoices(sortedVoices, true, false);
303 		setSecondVoices(sortedVoices, false, true);
304 		setSecondVoices(sortedVoices, false, false);
305 
306 		//log the available voices
307 		StringBuilder sb = new StringBuilder("Available voices:");
308 		for (Entry<Voice, TTSEngine> e : mBestEngines.entrySet()) {
309 			sb.append("\n* " + e.getKey() + " by "
310 			        + TTSServiceUtil.displayName(e.getValue().getProvider()));
311 		}
312 		ServerLogger.info(sb.toString());
313 		sb = new StringBuilder("Fallback voices:");
314 		for (Entry<Locale, Voice> e : mVoiceLangOnly.entrySet()) {
315 			sb.append("\n* " + e.getKey() + ": " + e.getValue());
316 		}
317 		ServerLogger.info(sb.toString());
318 	}

360 	private void registerVoice(VoiceInfo voiceInfo, Locale lang){
361 		registerVoice(mVoiceFullDescription, new VoiceKey(lang,
362 		        voiceInfo.gender, voiceInfo.voice.engine), voiceInfo.voice);
363 
364 		registerVoice(mVoiceGenderMissing, new VoiceKey(lang,
365 		        voiceInfo.voice.engine), voiceInfo.voice);
366 
367 		registerVoice(mVoiceEngineMissing, new VoiceKey(lang,
368 		        voiceInfo.gender), voiceInfo.voice);
369 
370 		registerVoice(mVoiceLangOnly, lang, voiceInfo.voice);
371 	}

373 	private <K> void registerVoice(Map<K, Voice> voiceMap, K key, Voice v) {
374 		if (!voiceMap.containsKey(key) && mBestEngines.containsKey(v))
375 			voiceMap.put(key, v);
376 	}

378 	private <K> Voice searchVoice(Map<K, Voice> voiceMap, K key1, K key2) {
379 		if (voiceMap.containsKey(key1))
380 			return voiceMap.get(key1);
381 
382 		return voiceMap.get(key2);
383 	}

385 	private Voice findAvailableVoice(String voiceEngine, String voiceName, String lang,
386 	        Gender gender, boolean[] exactMatch) {
387 
388 		if (voiceEngine != null && !voiceEngine.isEmpty() && voiceName != null
389 		        && !voiceName.isEmpty()) {
390 			Voice preferredVoice = new Voice(voiceEngine, voiceName);
391 			if (mBestEngines.containsKey(preferredVoice)) {
392 				exactMatch[0] = true;
393 				return preferredVoice;
394 			} else {
395 				Voice fallback = findSecondaryVoice(preferredVoice);
396 				if (fallback != null) {
397 					exactMatch[0] = false;
398 					return fallback;
399 				}
400 			}
401 		}
402 
403 		exactMatch[0] = false;
404 
405 		Locale loc;
406 		try {
407 			loc = VoiceInfo.tagToLocale(lang);
408 		} catch (UnknownLanguage e) {
409 			return null;
410 		}
411 		if (loc == null)
412 			return mBestMultiLangVoice;
413 
414 		Locale shortLoc = new Locale(loc.getLanguage());
415 		Voice result;
416 
417 		result = searchVoice(mVoiceFullDescription, new VoiceKey(loc, gender, voiceEngine),
418 		        new VoiceKey(shortLoc, gender, voiceEngine));
419 		if (result != null) {
420 			exactMatch[0] = (voiceName == null);
421 			return result;
422 		}
423 
424 		result = searchVoice(mVoiceGenderMissing, new VoiceKey(loc, voiceEngine),
425 		        new VoiceKey(shortLoc, voiceEngine));
426 		if (result != null) {
427 			exactMatch[0] = (voiceName == null && gender == null);
428 			return result;
429 		}
430 		result = searchVoice(mVoiceEngineMissing, new VoiceKey(loc, gender), new VoiceKey(
431 		        shortLoc, gender));
432 		if (result != null) {
433 			exactMatch[0] = (voiceName == null && voiceEngine == null);
434 			return result;
435 		}
436 
437 		exactMatch[0] = (voiceName == null && voiceEngine == null && gender == null);
438 		result = searchVoice(mVoiceLangOnly, loc, shortLoc);
439 		if (result != null){
440 			return result;
441 		}
442 		
443 		return mBestMultiLangVoice;
444 	}

446 	private void setSecondVoices(List<VoiceInfo> sortedAvailableVoices,
447 	        boolean sameEngine, boolean sameGender) {
448 		for (int i = 0; i < mVoicePriorities.size(); ++i) {
449 			VoiceInfo bestVoice = mVoicePriorities.get(i);
450 			if (!mSecondVoices.containsKey(bestVoice.voice)) {
451 				for (VoiceInfo fallback : sortedAvailableVoices) {
452 					if (!fallback.equals(bestVoice)){
453 						if (fallback.isMultiLang()){
454 							if (!sameGender && !sameEngine){
455 								//multilang fallback voices are only considered when gender and engine are not
456 								//criteria so as to prevent the algo from choosing a multilang voice with the
457 								//same engine over a regular voice with a different engine.
458 								mSecondVoices.put(bestVoice.voice, fallback.voice);
459 								break;
460 							}
461 						}
462 						else if (fallback.language.equals(bestVoice.language)
463 						        && (!sameGender || fallback.gender.equals(bestVoice.gender))
464 						        && (!sameEngine || fallback.voice.engine
465 						                .equals(bestVoice.voice.engine))) {
466 							mSecondVoices.put(bestVoice.voice, fallback.voice);
467 							break;
468 						}
469 					}
470 				}
471 			}
472 		}
473 	}

475 	private static String getStack(Throwable t) {
476 		StringWriter writer = new StringWriter();
477 		PrintWriter printWriter = new PrintWriter(writer);
478 		t.printStackTrace(printWriter);
479 		printWriter.flush();
480 		return writer.toString();
481 	}

505 			int res = this.lang.hashCode();

507 				res ^= this.gender.hashCode();
508 			if (this.engine != null)
509 				res ^= this.engine.hashCode();

512 		}
